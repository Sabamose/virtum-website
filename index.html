Получить API-ключ Документация
deploy-agent.ts 
import { VirtumClient } from 'virtum-js';

const client = new VirtumClient({
  apiKey: process.env.VIRTUM_API_KEY,
});

// Запускаем голосового агента
const agent = await client.phoneAgent.create({
  name: 'Аура',
  catalogId: 'cat_123',
  locationId: 'loc_456',
  capabilities: ['BOOKING', 'ORDERS'],
});

console.log('Номер агента:', agent.phoneNumber);
